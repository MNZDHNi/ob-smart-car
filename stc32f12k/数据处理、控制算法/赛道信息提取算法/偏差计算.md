// 基于左右电感值的偏差计算
float Calculate_Deviation(uint16_t left_val, uint16_t right_val)
{
    // 归一化处理
    float left_norm = (float)left_val / 4096.0f;
    float right_norm = (float)right_val / 4096.0f;
    
    // 差值法计算偏差
    float deviation = (left_norm - right_norm) / (left_norm + right_norm + 0.001f);
    
    return deviation;
}

// 多电感传感器融合偏差计算
float MultiSensor_Deviation(uint16_t *sensor_vals, uint8_t sensor_count)
{
    float weighted_sum = 0;
    float total_weight = 0;
    
    // 假设传感器等间距分布，中心为0
    for(int i=0; i<sensor_count; i++) {
        float position = (i - (sensor_count-1)/2.0f); // 位置权重
        float weight = sensor_vals[i] / 4096.0f;      // 信号强度权重
        weighted_sum += position * weight;
        total_weight += weight;
    }
    
    return weighted_sum / (total_weight + 0.001f);
}
