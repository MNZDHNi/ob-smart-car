typedef enum {
    TRACK_STRAIGHT,
    TRACK_CURVE,
    TRACK_CROSSROAD,
    TRACK_GARAGE,
    TRACK_RING
} TrackType;

TrackType Identify_Track(uint16_t *sensor_vals, uint8_t sensor_count)
{
    uint16_t max_val = 0;
    uint16_t min_val = 4096;
    uint8_t peak_count = 0;
    
    // 统计特征
    for(int i=0; i<sensor_count; i++) {
        if(sensor_vals[i] > max_val) max_val = sensor_vals[i];
        if(sensor_vals[i] < min_val) min_val = sensor_vals[i];
        if(sensor_vals[i] > 3000) peak_count++; // 阈值可调
    }
    
    // 判断逻辑
    if(peak_count >= 3) {
        return TRACK_CROSSROAD;    // 多个峰值，可能是十字
    } else if(max_val - min_val < 500) {
        return TRACK_STRAIGHT;     // 信号均匀，直道
    } else {
        return TRACK_CURVE;        // 有明显偏差，弯道
    }
}
